alphaTo 1 in 2 sec
$(".test").alphaTo(1,2);
slideTo 200,200 in 3 sec
$(".test").slideTo(200,200,3);
blurTo 40px, #F00 in 1 sec
$(".test").blurTo(40,"#F00",1);

then rotate back to 0deg in 3 sec

});

then rotate back to 0deg in 3 sec
$(".test").rotateY(360,3, "linear", function(){
$(".test").rotateY(0,3, "linear");
});
